 20          Example 
 21          ------- 
 22   
 23          >>> make_hapmapfile.py -f SNP_table.tab 
 24

 35                  - \-f <filename>, --fileIn <filename> 
 36                                                  Table of SNPs without FU 
 37   
 38          Input infos for running with default values: 
 39                  - \-o <filename>, --out <filename> 
 40                                                  Name of hapmap file (default outfile.hapmap) 
 41                  - \-w <int>, --window <int>

 66          # Parameters recovery

 69                                  description='''This program takes a SNP matrix with no missing data and returns a hapmap file. 
 70                                                          Files will be written in same directory as original file.''')

 72                                                  help='display make_hapmapfile.py version number and exit')

 74          filesreq = parser.add_argument_group('Input mandatory infos for running')

 76   
 77          files = parser.add_argument_group('Input infos for running with default values') 
 78          files.add_argument('-o', '--out', metavar="<filename>", default="outfile.hapmap", required = False, dest = 'fileOut', help = 'Name of hapmap file (default outfile.hapmap)') 
 79          files.add_argument('-w', '--window', metavar="<int>", type = int, default=1, required = False, dest = 'window', help = 'Minimal window by which SNPs have to be separated, in bp (default = 1 - keep everything)') 
 80          files.add_argument('-c', '--chrom', action='store_true', dest = 'chrom', help = 'If used, hapmap files will be produced by chromosomes') 
 81   
 82          # Check parameters 
 83          args = parser.parse_args() 
 84   
 85          # get arguments 
 86          fileIn = relativeToAbsolutePath(args.fileIn) 
 87          workdir = "/".join(fileIn.split("/")[:-1])+"/" 
 88          fileOut = workdir+args.fileOut 
 89          window = args.window 
 90          chrom = args.chrom 
 91   
 92          if window > 1: 
 93                  fileOut = fileOut.split(".")[0]+"_window"+str(window)+".hapmap" 
 94          if window == 0: 
 95                  window = 1

108          oldPos = 0 
109          nbSNP = 0

111          with open(fileIn) as noFUfile, open(fileOut, "w") as outFile: 
112                  header = noFUfile.readline().rstrip().split("\t") 
113                  headerOut = headerHapmap+"\t".join(header[3:])+"\n" 
114                  outFile.write(headerOut) 
115   
116                  # check numbers 
117                  SNPtot = 0 
118                  rsNumber = 0 
119                  removedLine = 0 
120                  nbLineTot = float(os.popen("wc -l "+fileIn).read().strip("\n").split(" ")[0]) 
121                  ctr = 0 
122   
123                  for line in noFUfile: 
124                          line = line.rstrip().split("\t") 
125                          curScaff = line[0] 
126                          SNPtot += 1 
127   
128                          # follow script progression on terminal 
129                          ctr += 1 
130                          if ((ctr % 10000 == 0) and (ctr != 0)) or (float(ctr) == nbLineTot): 
131                                  percent = (float(ctr)/float(nbLineTot))*100 
132                                  sys.stdout.write("\rProcessed up to %0.2f %%...\t" % percent) 
133                                  sys.stdout.flush() 
134   
135                          # check if scaffold name contains a number otherwise attribute 999 
136                          if len(re.findall(r'\d+', curScaff)) == 0: 
137                                  print("Scaffold '%s' contains no associated number, automatically attributing 999" %curScaff) 
138                                  scaffOut = 999 
139                          else:

141   
142                          if chrom == True:

145                                  if scaffOut not in dictFilesOut.keys(): 
146                                          dictFilesOut[scaffOut] = open(outputName, "w") 
147                                          dictFilesOut[scaffOut].write(headerOut)

149                                          sys.stdout.write("\n"+dictFilesOut[scaffOut].name.split("/")[-1])

151   
152                          # get position of SNP, reference allele and alleles for all samples 
153                          curPos = int(line[1]) 
154                          refAll = line[2] 
155                          listAll = line[3:] 
156                          listAll = [allele.replace('R', refAll) for allele in listAll] 
157   
158                          # check number of alleles at position 
159                          dictCount = {x:listAll.count(x) for x in listAll} 
160                          if chrom == True: dictSNPScaff[dictFilesOut[scaffOut]] = (dictSNPScaff.get(dictFilesOut[scaffOut], 0)) 
161   
162                          if curScaff != oldScaff: 
163                                  oldScaff = curScaff 
164                                  oldPos = 0 
165   
166                          diffPos = curPos - oldPos 
167   
168                          if diffPos <= window: 
169                                  continue 
170                          else: 
171                                  # keep SNP only if bi-allelic 
172                                  if len(dictCount.keys()) == 2: 
173                                          rsNumber += 1 
174   
175                                          if chrom == True: dictSNPScaff[dictFilesOut[scaffOut]] = (dictSNPScaff.get(dictFilesOut[scaffOut], 1))+1 
176   
177                                          # create allele couple 
178                                          alleleCouple = list(dictCount.keys()) 
179   
180                                          # create beginning of each line (without genotypes) 
181                                          lineHapmap = "rs%i\t%s/%s\t%s\t%s\t+\t2\tNA\tNA\tNA\tNA\tNA\t" %(rsNumber, alleleCouple[0], alleleCouple[1], scaffOut, curPos) 
182                                          # pseudo-diploidisation 
183                                          listPseudoDiploid = [allele+allele for allele in listAll] 
184                                          linePseudoDiploid = "\t".join(listPseudoDiploid) 
185                                          # complete line 
186                                          lineFileOut = lineHapmap+linePseudoDiploid+"\n" 
187   
188                                          if chrom == True: dictFilesOut[scaffOut].write(lineFileOut) 
189                                          outFile.write(lineFileOut) 
190   
191                                  else: 
192                                          removedLine += 1 
193                                  oldPos = curPos 
194   
195                  sys.stdout.flush() 
196   
197                  if chrom == True: 
198                          print("\n\n#### CLOSING FILES") 
199                          for keys in sorted(dictFilesOut.keys()): 
200                                  openFiles = dictFilesOut[keys] 
201                                  print("%s\t(%i SNPs)" %(openFiles.name.split("/")[-1], dictSNPScaff[openFiles])) 
202                                  openFiles.close() 
203   
204                  print("\n%i SNPs in the original file." %SNPtot) 
205                  print("%i SNPs successfully written in Hapmap format." %rsNumber)
